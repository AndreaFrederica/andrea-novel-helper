<template>
  <div>
    <div style="height: calc(100vh)">
      <RelationGraph ref="graphRef" :options="graphOptions">
        <template #tool-bar>
          <ObjectEditToolBar />
        </template>

<script lang="ts" setup>
import { defineComponent, onMounted, ref } from 'vue';
import RelationGraph, { RGJsonData, RGOptions, RGNode, RGLine, RGLink, RGUserEvent, RelationGraphComponent } from 'relation-graph-vue3';
import ObjectEditToolBar from './VipObjectEditToolBar';

const graphOptions: RGOptions = {
    debug: false,
    allowSwitchLineShape: true,
    allowSwitchJunctionPoint: true,
    allowShowDownloadButton: true,
    defaultJunctionPoint: 'border'
};

const graphRef = ref<RelationGraphComponent>();

onMounted(() => {
    showGraph();
});

const showGraph = async() => {
    const __graph_json_data: RGJsonData = {
        rootId: 'a',
        nodes: [
            { id: 'a', text: 'A', borderColor: 'yellow' },
            { id: 'b', text: 'B', color: '#43a2f1', fontColor: 'yellow' },
            { id: 'c', text: 'C', nodeShape: 1, width: 80, height: 60 },
            { id: 'e', text: 'E', nodeShape: 0, width: 150, height: 150 }
        ],
        lines: [
            { from: 'a', to: 'b', text: 'Relation 1', color: '#43a2f1' },
            { from: 'a', to: 'c', text: 'Relation 2' },
            { from: 'a', to: 'e', text: 'Relation 3' },
            { from: 'b', to: 'e', text: '', color: '#67C23A' }
        ]
    };

    const graphInstance = graphRef.value?.getInstance();
    if (graphInstance) {
        await graphInstance.setJsonData(__graph_json_data);
        await graphInstance.moveToCenter();
        await graphInstance.zoomToFit();
    }
};
</script>

<template>
  <div style="position: absolute;z-index: 30;top:10px;left: 40px; padding:10px;width:420px;height:50px;background-color: #ffffff;border:#efefef solid 1px;box-shadow: 0 3px 9px rgba(0,21,41,.08);border-radius: 10px;">

      <div class="c-mb-button" @click="onSave">
        <CircumIcon size="20px" name="cloud_on" />
        <div class="c-mb-text">Save</div>
      </div>
      <div class="c-mb-button" style="line-height:40px;font-size: 12px;text-align: center;" @click="zoomToFit">
        {{ options.canvasZoom }}%
      </div>
      <div class="c-mb-button c-mb-button-c" :class="{'c-mb-button-on':options.creatingNodePlot}" style="width: 100px;" @click="startAddNode">
        <CircumIcon size="20px" name="circle_plus" />
        <div class="c-mb-text">Node(Click)</div>
      </div>
      <div class="c-mb-button c-mb-button-c" :class="{'c-mb-button-on':options.creatingNodePlot}" style="width: 100px;" @mousedown="startAddNode">
        <CircumIcon size="20px" name="circle_plus" />
        <div class="c-mb-text">Node(Drag)</div>
      </div>
      <div class="c-mb-button c-mb-button-c" :class="{'c-mb-button-on':options.creatingLinePlot}" style="width: 50px;" @click="startAddLine">
        <CircumIcon size="20px" name="route" />
        <div class="c-mb-text">Line</div>
      </div>
      <div class="c-mb-button" @click="zoomToFit">
        <CircumIcon size="20px" name="maximize_2" />
        <div class="c-mb-text">Fit</div>
      </div>
      <div class="c-mb-button" @click="refresh">
        <CircumIcon size="20px" name="undo" />
        <div class="c-mb-text">Refresh</div>
      </div>
    </div>
  </div>
</template>

<script lang="ts" setup>
import {computed, defineComponent, inject, ref} from 'vue';
import type { RGOptions, RGNode, RGLine, RGUserEvent, RGJsonData, RelationGraphComponent } from 'relation-graph-vue3';
import {ElNotification} from 'element-plus';
import CircumIcon from '@klarr-agency/circum-icons-vue'; // Vue
import {graphKey} from 'relation-graph-vue3';
import {RelationGraphFinal} from 'relation-graph/types/types/models/RelationGraphFinal';

const showNewNodeTemplate = ref(false);
const newObjectTemplatePosition = ref({x: 0, y: 0});
const height = ref(275);
const newNodeIdIndex = ref(1);
const newLineIdIndex = ref(1);
const graph = inject(graphKey);
const options = computed(() => {
    return graph && graph.options;
});
const relationGraph = computed(() => {
    return graph && graph.instance;
});
const about = () => {
    alert('Welcome to Apple Products !');
};
const refresh = () => {
    const graphInstance = relationGraph.value;
    graphInstance.refresh();
};
const switchLayout = async(layoutConfig: any) => {
    const graphInstance = relationGraph.value;
    await graphInstance.switchLayout(layoutConfig, options.value);
    await graphInstance.refresh();
};
const toggleAutoLayout = () => {
    const graphInstance = relationGraph.value;
    graphInstance.toggleAutoLayout();
};
const downloadAsFile = (format: string) => {
    alert('download as :' + format);
};
const zoomToFit = async () => {
    const graphInstance = relationGraph.value;
    await graphInstance.setZoom(100);
    await graphInstance.moveToCenter();
    await graphInstance.zoomToFit();
};
const startAddNode = (e: MouseEvent) => {
    const graphInstance = relationGraph.value;
    graphInstance.startCreatingNodePlot(e, {
        templateText: 'Node',
        templateNode: {
            className: 'my-node-template'
        },
        onCreateNode: (x: number, y: number) => {
            console.log('New node:', x, y);
            const newId = newNodeIdIndex.value++;
            relationGraph.value.addNodes([{
                id: 'newNode-' + newId,
                text: 'New Node ' + newId,
                color: '#5da0f8',
                x: x - 50,
                y: y - 50

            }]);
        }
    });
};
const startAddLine = (e: MouseEvent) => {
    ElNotification({type: 'success', message: 'Click on a node to start creating a line!'});
    const graphInstance = relationGraph.value;
    graphInstance.startCreatingLinePlot(e, {
        template: {
            lineWidth: 3,
            color: '#8080ff',
            text: 'New Line'
        },
        onCreateLine: (from: RGNode, to: RGNode) => {
            console.log('New line:', from, to);
            if (to.id) {
                const newLineId = newLineIdIndex.value++;
                relationGraph.value.addLines([{
                    from: from.id,
                    to: to.id,
                    lineWidth: 3,
                    color: '#8080ff',
                    text: 'New Line ' + newLineId

                }]);
            }
        }
    });
};

</script>
